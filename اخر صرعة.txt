        function calculateInitialZoom() {
            const scrollArea = document.getElementById('preview-scroll-area');
            const containerWidth = 1123;
            const containerHeight = 794;
            const availableWidth = scrollArea.clientWidth - 64; 
            const availableHeight = scrollArea.clientHeight - 64;
            scale = Math.min(availableWidth / containerWidth, availableHeight / containerHeight, 1);
            applyZoom();
        }

        function zoomPreview(delta) {

        function handleLogoUpload(event) {
            const file = event.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    uploadedLogo = e.target.result;
                    document.getElementById('input-logo-url').value = '';
                    updatePreview();
                };
                reader.readAsDataURL(file);
            }
        }

        function handleLogoUrl() {

        async function generateCanvasForExport() {
            const scrollArea = document.getElementById('preview-scroll-area');
            const scaleWrapper = document.getElementById('scale-wrapper');
            const element = document.getElementById('certificate-container');
            
            const originalTransform = scaleWrapper.style.transform;
            const originalScrollTop = scrollArea.scrollTop;
            const originalScrollLeft = scrollArea.scrollLeft;
            
            scaleWrapper.style.transform = 'scale(1)';
            scrollArea.scrollTop = 0;
            scrollArea.scrollLeft = 0;
            
            await new Promise(r => setTimeout(r, 400)); 
            
            // قراءة لون الخلفية الحالي عند التصدير
            let exportBg = document.getElementById('color-bg').value;

            const canvas = await html2canvas(element, {
                scale: 2, 
                useCORS: true,
                allowTaint: true,
                backgroundColor: exportBg,
                width: 1123,
                height: 794
            });
            
            scaleWrapper.style.transform = originalTransform;
            scrollArea.scrollTop = originalScrollTop;
            scrollArea.scrollLeft = originalScrollLeft;
            
            return canvas;
        }

        async function downloadPNG() {
            const rawNames = document.getElementById('input-names').value;
            const names = rawNames.split('\n').map(n => n.trim()).filter(n => n !== '');
            if(names.length > 1) {
                showToast("تحميل الصورة (PNG) يدعم طالباً واحداً فقط. لتحميل قائمة الطلاب، يرجى استخدام (PDF).", "error");
                return;
            }

            showLoader('جاري تجهيز الصورة بجودة عالية...');
            
            try {
                updatePreview();
                const canvas = await generateCanvasForExport();
                
                const link = document.createElement('a');
                link.download = `شهادة_${names.length > 0 ? names[0] : 'فارغة'}.png`;
                link.href = canvas.toDataURL('image/png', 1.0);
                link.click();
            } catch (error) {
                console.error("Error generating PNG:", error);
                showToast("حدث خطأ أثناء إنشاء الصورة.", "error");
            } finally {
                hideLoader();
            }
        }

        async function downloadPDF() {
            const rawNames = document.getElementById('input-names').value;
            const names = rawNames.split('\n').map(n => n.trim()).filter(n => n !== '');
            
            const { jsPDF } = window.jspdf;
            const pdf = new jsPDF({ orientation: 'landscape', unit: 'mm', format: 'a4' });
            const processNames = names.length > 0 ? names : [null];

            showLoader(processNames.length > 1 ? `جاري إصدار ${processNames.length} شهادات...` : 'جاري تجهيز ملف PDF...');

            try {
                for (let i = 0; i < processNames.length; i++) {
                    updatePreview(processNames[i]);
                    const canvas = await generateCanvasForExport();
                    const imgData = canvas.toDataURL('image/jpeg', 0.98); 

                    if (i > 0) pdf.addPage();
                    pdf.addImage(imgData, 'JPEG', 0, 0, 297, 210);
                }

                pdf.save(processNames.length > 1 ? 'الشهادات_المجمعة.pdf' : `شهادة_${processNames[0] || 'فارغة'}.pdf`);
            } catch (error) {
                console.error("Error generating PDF:", error);
                showToast("حدث خطأ أثناء إنشاء ملف PDF.", "error");
            } finally {
                updatePreview();
                hideLoader();
            }
        }

        function showLoader(text) {
            document.getElementById('loader-text').innerText = text;
            document.getElementById('loader').style.display = 'flex';
        }

        function hideLoader() {
            document.getElementById('loader').style.display = 'none';
        }

        window.onload = () => {
            calculateInitialZoom();
            resetColors(); 
        };
